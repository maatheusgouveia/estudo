CREATE DATABASE EXERCICIOSGBD;

CREATE TABLE CLIENTE(
	CLIENTEID INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
	CLIENTENOME VARCHAR(30) NOT NULL
);

CREATE TABLE VEICULO(
	VEICULOID INT PRIMARY KEY NOT NULL AUTO_INCREMENT,
    	VEICULOMODELO VARCHAR(20),
    	VEICULOPLACA VARCHAR(10),
    	CLIENTEID INT,
    	FOREIGN KEY(CLIENTEID) REFERENCES CLIENTE(CLIENTEID)
);
